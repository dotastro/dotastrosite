<script>
// Refresh bios and avatars from live Bluesky API
(function () {
  var cards = document.querySelectorAll('.person-card[data-handle]');
  cards.forEach(function (card) {
    var handle = card.getAttribute('data-handle');
    fetch('https://public.api.bsky.app/xrpc/app.bsky.actor.getProfile?actor=' + handle)
      .then(function (r) { return r.json(); })
      .then(function (d) {
        // Update avatar
        var img = card.querySelector('.person-avatar');
        if (img && d.avatar) {
          img.src = d.avatar + '@jpeg';
          img.alt = d.displayName || handle;
        }
        // Update bio
        var bio = card.querySelector('[data-bsky-bio]');
        if (bio && d.description) {
          // Strip URLs from bio text for cleanliness
          var cleaned = d.description.replace(/https?:\/\/\S+/g, '').trim();
          if (cleaned.length > 20) bio.textContent = cleaned;
        }
      })
      .catch(function () { /* fail silently, static content already shown */ });
  });
})();
</script>
